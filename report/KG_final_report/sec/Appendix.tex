\section{Selected SPARQL Queries and Results}
\label{sec:sparql}

All queries share the following prefix block (omitted from individual listings):
\begin{footnotesize}
\begin{verbatim}
PREFIX mrc: <http://purl.org/ontology/mrc/>
PREFIX mo:  <http://purl.org/ontology/mo/>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX wdt: <http://www.wikidata.org/prop/direct/>
PREFIX foaf: <http://xmlns.com/foaf/0.1/>
PREFIX dct: <http://purl.org/dc/terms/>
\end{verbatim}
\end{footnotesize}

\subsection*{Q1 — Genre Ranking (Simple KG)}
\begin{footnotesize}
\begin{verbatim}
SELECT ?genreLabel (COUNT(DISTINCT ?artist) AS ?n)
WHERE { ?artist mrc:hasGenre ?g .
        ?g rdfs:label ?genreLabel .
        FILTER(LANG(?genreLabel) = "en") }
GROUP BY ?genreLabel  ORDER BY DESC(?n)  LIMIT 30
\end{verbatim}
\end{footnotesize}

\begin{table}[H]\centering\footnotesize
\caption{Top 5 genres by artist count (Simple KG).}
\begin{tabular}{lr}\toprule Genre & \#Artists\\\midrule
rock & 2,117 \\ pop & 1,062 \\ electronic & 1,037 \\ pop rock & 814 \\ hip hop & 381\\\bottomrule
\end{tabular}
\end{table}

\subsection*{Q2 — Genre Weights (Rich KG)}
\begin{footnotesize}
\begin{verbatim}
SELECT ?genreLabel (COUNT(DISTINCT ?artist) AS ?n)
  (ROUND(AVG(?w)*1000)/1000 AS ?avg_weight)
WHERE { ?artist mrc:hasGenreAssoc ?a .
        ?a mrc:genre ?g; mrc:weight ?w .
        ?g rdfs:label ?genreLabel .
        FILTER(LANG(?genreLabel) = "en") }
GROUP BY ?genreLabel  ORDER BY DESC(?avg_weight)  LIMIT 10
\end{verbatim}
\end{footnotesize}

\noindent Top by average weight: dance pop (0.810), trance (0.795), ballad (0.779), rock (0.759).

\subsection*{Q3 — High-Confidence Key Distribution (Rich KG)}
\begin{footnotesize}
\begin{verbatim}
SELECT ?keyLabel (COUNT(*) AS ?n)
WHERE { ?p a mo:Performance; mrc:hasKey ?k;
           mrc:keyConfidence ?kc .
        FILTER(?kc >= 0.8)
        ?k rdfs:label ?keyLabel .
        FILTER(LANG(?keyLabel) = "en") }
GROUP BY ?keyLabel  ORDER BY DESC(?n)
\end{verbatim}
\end{footnotesize}

\noindent Top keys at $\geq$0.8 confidence: C\,(211), G\,(196), D\,(150), A\,(103), F\,(98).
